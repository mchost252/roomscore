import React, { useRef, useLayoutEffect, useState, useCallback, useEffect } from 'react';
import {
  View, Text, StyleSheet, TouchableOpacity, ScrollView, Animated,
  RefreshControl, TextInput, Modal, ActivityIndicator,
} from 'react-native';
import { useRouter } from 'expo-router';
import { Ionicons } from '@expo/vector-icons';
import { LinearGradient } from 'expo-linear-gradient';
import { useTheme } from '../../context/ThemeContext';
import { useAuth } from '../../context/AuthContext';
import { useToast } from '../../context/ToastContext';
import { useSafeAreaInsets } from 'react-native-safe-area-context';
import { useHaptics } from '../../hooks';
import roomService from '../../services/roomService';
import { Room } from '../../types';
import Skeleton, { SkeletonGroup } from '../../components/ui/Skeleton';

type Tab = 'mine' | 'discover';

export default function RoomsScreen() {
  const router = useRouter();
  const { colors, gradients, isDark } = useTheme();
  const { user } = useAuth();
  const { showToast } = useToast();
  const haptics = useHaptics();
  const insets = useSafeAreaInsets();

  // ── Entrance animation ──
  const fadeAnim = useRef(new Animated.Value(0)).current;
  const slideAnim = useRef(new Animated.Value(20)).current;
  useLayoutEffect(() => {
    Animated.parallel([
      Animated.timing(fadeAnim, { toValue: 1, duration: 200, useNativeDriver: true }),
      Animated.spring(slideAnim, { toValue: 0, tension: 220, friction: 18, useNativeDriver: true }),
    ]).start();
  }, []);

  // ── Theme ──
  const t = {
    bg: colors.background.primary,
    surface: colors.surface,
    border: colors.border.primary,
    text: colors.text,
    textSub: colors.textSecondary,
    textHint: colors.textTertiary,
    primary: colors.primary,
    gradient: gradients.background.colors,
  };

  // ── State ──
  const [activeTab, setActiveTab] = useState<Tab>('mine');
  const [myRooms, setMyRooms] = useState<Room[]>([]);
  const [publicRooms, setPublicRooms] = useState<Room[]>([]);
  const [loading, setLoading] = useState(true);
  const [refreshing, setRefreshing] = useState(false);
  const [joinModalVisible, setJoinModalVisible] = useState(false);
  const [joinCode, setJoinCode] = useState('');
  const [joining, setJoining] = useState(false);

  // ── Data loading ──
  const loadRooms = useCallback(async (isRefresh = false) => {
    try {
      if (!isRefresh) setLoading(true);
      if (activeTab === 'mine') {
        const rooms = await roomService.getMyRooms();
        setMyRooms(rooms);
      } else {
        const rooms = await roomService.getPublicRooms();
        setPublicRooms(rooms);
      }
    } catch (err: any) {
      showToast({ message: err?.message || 'Failed to load rooms', type: 'error' });
    } finally {
      setLoading(false);
      setRefreshing(false);
    }
  }, [activeTab]);

  useEffect(() => {
    loadRooms();
  }, [activeTab]);

  const onRefresh = useCallback(() => {
    setRefreshing(true);
    loadRooms(true);
  }, [loadRooms]);

  // ── Join by code ──
  const handleJoin = useCallback(async () => {
    const code = joinCode.trim();
    if (!code) return;
    setJoining(true);
    try {
      const result = await roomService.joinRoom(code);
      haptics.success();
      setJoinModalVisible(false);
      setJoinCode('');
      if (result.pending) {
        showToast({ message: 'Join request sent! Waiting for approval.', type: 'info' });
      } else if (result.room) {
        showToast({ message: `Joined ${result.room.name}!`, type: 'success' });
        router.push({ pathname: '/(home)/room-detail', params: { roomId: result.room.id } });
      }
      // Refresh my rooms list
      if (activeTab === 'mine') loadRooms(true);
    } catch (err: any) {
      haptics.error();
      showToast({ message: err?.response?.data?.message || 'Failed to join room', type: 'error' });
    } finally {
      setJoining(false);
    }
  }, [joinCode, activeTab]);

  // ── Join public room ──
  const handleJoinPublic = useCallback(async (room: Room) => {
    try {
      const result = await roomService.joinRoom(room.joinCode);
      haptics.success();
      if (result.pending) {
        showToast({ message: 'Join request sent! Waiting for approval.', type: 'info' });
      } else {
        showToast({ message: `Joined ${room.name}!`, type: 'success' });
        setActiveTab('mine');
      }
      loadRooms(true);
    } catch (err: any) {
      haptics.error();
      showToast({ message: err?.response?.data?.message || 'Failed to join', type: 'error' });
    }
  }, []);

  // ── Helpers ──
  const getCompletedCount = (room: Room) => room.tasks?.filter(t => t.isCompleted).length || 0;
  const getTotalTasks = (room: Room) => room.tasks?.length || 0;
  const getMemberCount = (room: Room) => room.members?.length || 0;
  const isOwner = (room: Room) => room.ownerId === user?.id || room.owner?._id === user?.id;

  // ── Render room card using new enhanced component ──
  const renderRoomCard = (room: Room, isPublic = false) => {
    // Convert Room to EnhancedRoom format for the new component
    const enhancedRoom = {
      id: room.id || room._id,
      name: room.name,
      code: room.joinCode,
      ownerId: room.ownerId,
      createdAt: new Date(room.createdAt).getTime(),
      expiresAt: room.endDate ? new Date(room.endDate).getTime() : Date.now() + 30 * 24 * 60 * 60 * 1000,
      isPremium: room.isPremium,
      memberCount: room.members?.length || 0,
      taskCount: room.tasks?.length || 0,
      isFlipped: false,
    };

    return (
      <RoomCard
        room={enhancedRoom}
        onPress={() => {
          if (isPublic) return;
          haptics.tap();
          router.push({ pathname: '/(home)/room-detail', params: { roomId: room.id || room._id } });
        }}
        isFirstEntry={true}
        index={0}
      />
    );
  };

  // ── Loading skeleton ──
  const renderSkeleton = () => (
    <View style={{ paddingHorizontal: 20, gap: 12 }}>
      {[1, 2, 3].map(i => (
        <View key={i} style={[styles.roomCard, { backgroundColor: t.surface, borderColor: t.border }]}>
          <SkeletonGroup gap={10}>
            <View style={{ flexDirection: 'row', alignItems: 'center', gap: 12 }}>
              <Skeleton width={40} height={40} borderRadius={12} />
              <View style={{ flex: 1, gap: 6 }}>
                <Skeleton width="60%" height={16} borderRadius={6} />
                <Skeleton width="40%" height={12} borderRadius={4} />
              </View>
            </View>
            <Skeleton width="100%" height={4} borderRadius={2} />
          </SkeletonGroup>
        </View>
      ))}
    </View>
  );

  // ── Empty state ──
  const renderEmpty = () => (
    <View style={styles.emptyContainer}>
      <LinearGradient
        colors={['#6366f1', '#8b5cf6']}
        start={{ x: 0, y: 0 }}
        end={{ x: 1, y: 1 }}
        style={styles.emptyIcon}
      >
        <Ionicons name={activeTab === 'mine' ? 'people' : 'search'} size={32} color="#fff" />
      </LinearGradient>
      <Text style={[styles.emptyTitle, { color: t.text }]}>
        {activeTab === 'mine' ? 'No Rooms Yet' : 'No Public Rooms'}
      </Text>
      <Text style={[styles.emptySubtitle, { color: t.textSub }]}>
        {activeTab === 'mine'
          ? 'Create a room or join one with a code to start building habits together.'
          : 'No public rooms available right now. Check back later or create your own!'}
      </Text>
      {activeTab === 'mine' && (
        <View style={styles.emptyActions}>
          <TouchableOpacity
            activeOpacity={0.8}
            onPress={() => {
              haptics.tap();
              router.push('/(home)/create-room');
            }}
            style={{ borderRadius: 14, overflow: 'hidden' }}
          >
            <LinearGradient
              colors={['#6366f1', '#8b5cf6']}
              start={{ x: 0, y: 0 }}
              end={{ x: 1, y: 0 }}
              style={styles.emptyBtn}
            >
              <Ionicons name="add" size={16} color="#fff" />
              <Text style={styles.emptyBtnText}>Create Room</Text>
            </LinearGradient>
          </TouchableOpacity>
          <TouchableOpacity
            activeOpacity={0.8}
            onPress={() => {
              haptics.tap();
              setJoinModalVisible(true);
            }}
            style={[styles.emptyBtnOutline, { borderColor: t.border }]}
          >
            <Ionicons name="enter-outline" size={16} color={t.primary} />
            <Text style={[styles.emptyBtnOutlineText, { color: t.primary }]}>Join with Code</Text>
          </TouchableOpacity>
        </View>
      )}
    </View>
  );

  const rooms = activeTab === 'mine' ? myRooms : publicRooms;

  return (
    <Animated.View style={{ flex: 1, opacity: fadeAnim, transform: [{ translateY: slideAnim }] }}>
      <View style={[styles.container, { backgroundColor: t.bg }]}>
        <LinearGradient
          colors={t.gradient as any}
          start={{ x: 0.2, y: 0 }}
          end={{ x: 0.8, y: 1 }}
          style={StyleSheet.absoluteFill}
        />

        {/* ── Header ── */}
        <View style={[styles.header, { paddingTop: Math.max(insets.top + 8, 52) }]}>
          <TouchableOpacity
            onPress={() => router.back()}
            style={[styles.backButton, { backgroundColor: t.surface, borderColor: t.border }]}
          >
            <Ionicons name="chevron-back" size={22} color={t.text} />
          </TouchableOpacity>
          <Text style={[styles.title, { color: t.text }]}>Rooms</Text>
          <TouchableOpacity
            onPress={() => {
              haptics.tap();
              router.push('/(home)/create-room');
            }}
            style={[styles.backButton, { backgroundColor: t.surface, borderColor: t.border }]}
          >
            <Ionicons name="add" size={22} color={t.primary} />
          </TouchableOpacity>
        </View>

        {/* ── Tab bar ── */}
        <View style={[styles.tabBar, { borderColor: t.border }]}>
          <TouchableOpacity
            style={[styles.tab, activeTab === 'mine' && styles.tabActive, activeTab === 'mine' && { borderColor: t.primary }]}
            onPress={() => { haptics.selection(); setActiveTab('mine'); }}
            activeOpacity={0.7}
          >
            <Ionicons name="home-outline" size={16} color={activeTab === 'mine' ? t.primary : t.textSub} />
            <Text style={[styles.tabText, { color: activeTab === 'mine' ? t.primary : t.textSub }]}>
              My Rooms
            </Text>
          </TouchableOpacity>
          <TouchableOpacity
            style={[styles.tab, activeTab === 'discover' && styles.tabActive, activeTab === 'discover' && { borderColor: t.primary }]}
            onPress={() => { haptics.selection(); setActiveTab('discover'); }}
            activeOpacity={0.7}
          >
            <Ionicons name="compass-outline" size={16} color={activeTab === 'discover' ? t.primary : t.textSub} />
            <Text style={[styles.tabText, { color: activeTab === 'discover' ? t.primary : t.textSub }]}>
              Discover
            </Text>
          </TouchableOpacity>
          <TouchableOpacity
            style={[styles.tab]}
            onPress={() => { haptics.tap(); setJoinModalVisible(true); }}
            activeOpacity={0.7}
          >
            <Ionicons name="enter-outline" size={16} color={t.textSub} />
            <Text style={[styles.tabText, { color: t.textSub }]}>Join Code</Text>
          </TouchableOpacity>
        </View>

        {/* ── Content ── */}
        {loading ? renderSkeleton() : rooms.length === 0 ? (
          <ScrollView
            contentContainerStyle={styles.scrollContent}
            refreshControl={<RefreshControl refreshing={refreshing} onRefresh={onRefresh} tintColor={t.primary} />}
          >
            {renderEmpty()}
          </ScrollView>
        ) : (
          <ScrollView
            style={styles.content}
            contentContainerStyle={styles.scrollContent}
            showsVerticalScrollIndicator={false}
            refreshControl={<RefreshControl refreshing={refreshing} onRefresh={onRefresh} tintColor={t.primary} />}
          >
            {rooms.map(room => renderRoomCard(room, activeTab === 'discover'))}
          </ScrollView>
        )}
      </View>

      {/* ── Join Code Modal ── */}
      <Modal visible={joinModalVisible} transparent animationType="fade" onRequestClose={() => setJoinModalVisible(false)}>
        <View style={styles.modalOverlay}>
          <View style={[styles.modalContent, { backgroundColor: isDark ? '#1a1a2e' : '#fff', borderColor: t.border }]}>
            <View style={styles.modalHeader}>
              <Text style={[styles.modalTitle, { color: t.text }]}>Join a Room</Text>
              <TouchableOpacity onPress={() => setJoinModalVisible(false)}>
                <Ionicons name="close" size={24} color={t.textSub} />
              </TouchableOpacity>
            </View>

            <Text style={[styles.modalSubtitle, { color: t.textSub }]}>
              Enter the invite code shared by the room owner.
            </Text>

            <TextInput
              style={[styles.codeInput, {
                backgroundColor: isDark ? 'rgba(255,255,255,0.06)' : 'rgba(0,0,0,0.04)',
                borderColor: t.border,
                color: t.text,
              }]}
              placeholder="e.g. ABCD1234"
              placeholderTextColor={t.textHint}
              value={joinCode}
              onChangeText={(text) => setJoinCode(text.toUpperCase())}
              autoCapitalize="characters"
              autoCorrect={false}
              maxLength={12}
              returnKeyType="go"
              onSubmitEditing={handleJoin}
            />

            <TouchableOpacity
              activeOpacity={0.8}
              onPress={handleJoin}
              disabled={joining || !joinCode.trim()}
              style={{ borderRadius: 14, overflow: 'hidden', opacity: joining || !joinCode.trim() ? 0.5 : 1 }}
            >
              <LinearGradient
                colors={['#6366f1', '#8b5cf6']}
                start={{ x: 0, y: 0 }}
                end={{ x: 1, y: 0 }}
                style={styles.modalJoinBtn}
              >
                {joining ? (
                  <ActivityIndicator size="small" color="#fff" />
                ) : (
                  <>
                    <Ionicons name="enter-outline" size={16} color="#fff" />
                    <Text style={styles.modalJoinBtnText}>Join Room</Text>
                  </>
                )}
              </LinearGradient>
            </TouchableOpacity>
          </View>
        </View>
      </Modal>
    </Animated.View>
  );
}
